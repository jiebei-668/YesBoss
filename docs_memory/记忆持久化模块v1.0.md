# YesBoss记忆持久化模块 v1.0 - 完整接口设计

## 概述

本文档定义了YesBoss记忆持久化模块的完整接口设计，采用三层架构（Resource、Snippet、Preference），支持两种向量存储后端（sqlite-vec和PostgreSQL+pgvector），通过配置文件灵活切换。

---

## 架构概览

```
┌─────────────────────────────────────────────────┐
│         MemoryQueryService（统一查询）            │
├─────────────────────────────────────────────────┤
│  MemoryManager（三层关联管理）                    │
├──────────────┬──────────────┬───────────────────┤
│  Resource    │ Snippet  │    Preference       │
│  Repository  │  Repository  │   Repository      │
│  (原始数据)  │ (结构化记忆)  │   (分类主题)      │
├──────────────┴──────────────┴───────────────────┤
│         VectorStore（向量存储抽象）               │
├──────────────────┬──────────────────────────────┤
│  SQLiteVecStore  │  PostgreSQLPgVectorStore     │
└──────────────────┴──────────────────────────────┘
```

package tech.yesboss.memory.vector;

import java.util.List;
import java.util.Map;

/**
 * 向量存储抽象接口
 * 支持多种向量数据库后端（sqlite-vec, PostgreSQL+pgvector）
 */
public interface VectorStore {

    /**
     * 创建向量索引
     * @param indexName 索引名称
     * @param dimensions 向量维度
     */
    void createIndex(String indexName, int dimensions);

    /**
     * 插入向量数据
     * @param indexName 索引名称
     * @param id 数据ID
     * @param vector 向量数据
     * @param metadata 元数据
     */
    void insert(String indexName, String id, float[] vector, Map<String, Object> metadata);

    /**
     * 批量插入向量数据
     * @param indexName 索引名称
     * @param vectorDataList 向量数据列表
     */
    void bulkInsert(String indexName, List<VectorData> vectorDataList);

    /**
     * 向量相似度搜索
     * @param indexName 索引名称
     * @param queryVector 查询向量
     * @param topK 返回前K个结果
     * @return 搜索结果列表
     */
    List<VectorSearchResult> search(String indexName, float[] queryVector, int topK);

    /**
     * 带过滤条件的向量搜索
     * @param indexName 索引名称
     * @param queryVector 查询向量
     * @param topK 返回前K个结果
     * @param filters 过滤条件
     * @return 搜索结果列表
     */
    List<VectorSearchResult> search(String indexName, float[] queryVector, int topK, Map<String, Object> filters);

    /**
     * 更新向量数据
     * @param indexName 索引名称
     * @param id 数据ID
     * @param vector 向量数据
     * @param metadata 元数据
     */
    void update(String indexName, String id, float[] vector, Map<String, Object> metadata);

     * 删除向量数据
     * @param indexName 索引名称
     * @param id 数据ID
     */
    void delete(String indexName, String id);

     * 批量删除向量数据
     * @param indexName 索引名称
     * @param ids 数据ID列表

    void bulkDelete(String indexName, List<String> ids);

     * 获取向量数据
     * @param indexName 索引名称
     * @param id 数据ID
     * @return 向量数据

    VectorData get(String indexName, String id);

    /**
     * 获取索引信息
     * @param indexName 索引名称
     * @return 索引信息
     */

    /**
     * 检查索引是否存在
     * @param indexName 索引名称
     * @return 是否存在
     */

    /**
     * 删除索引

 */
public enum VectorStoreType {
    SQLITE_VEC("sqlite-vec"),
    POSTGRESQL_PGVECTOR("postgresql-pgvector");

    private final String value;

    VectorStoreType(String value) {
        this.value = value;
    }

    public String getValue() {
        return value;
    }

    public static VectorStoreType fromValue(String value) {
        for (VectorStoreType type : values()) {

                return type;

        }
        throw new IllegalArgumentException("Unknown vector store type: " + value);
